<!-- Pointer Events API for CFIX-02 -->
```typescript
// Replace mouse handlers with pointer handlers:
function startDraw(e: React.PointerEvent<HTMLCanvasElement>) {
  if (signed) return;
  e.currentTarget.setPointerCapture(e.pointerId);
  isDrawing.current = true;
  const canvas = e.currentTarget;
  const ctx = canvas.getContext('2d')!;
  const rect = canvas.getBoundingClientRect();
  ctx.beginPath();
  ctx.moveTo(e.clientX - rect.left, e.clientY - rect.top);
}
// draw(), stopDraw() use same e.clientX/clientY (PointerEvent extends MouseEvent)

// Canvas element changes:
// onMouseDown={startDraw} → onPointerDown={startDraw}
// onMouseMove={draw}      → onPointerMove={draw}
// onMouseUp={stopDraw}    → onPointerUp={stopDraw}
// onMouseLeave={stopDraw} → onPointerLeave={stopDraw}
```

```typescript
// From @monobase/sdk-ts/generated/react-query:
import { listLabOrdersOptions, listLabOrdersQueryKey, createLabOrderMutation, updateLabOrderMutation } from '@monobase/sdk-ts/generated/react-query';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

// List (replaces useEffect + load()):
const { data: ordersResponse, isLoading: loading } = useQuery(
  listLabOrdersOptions({ path: { visitId } })
);
const orders = (ordersResponse?.data ?? []) as LabOrder[];

// Mutations:
const queryClient = useQueryClient();
const invalidate = () => queryClient.invalidateQueries({ queryKey: listLabOrdersQueryKey({ path: { visitId } }) });

const updateMutation = useMutation({
  ...updateLabOrderMutation(),
  onSuccess: invalidate,
});
const createMutation = useMutation({
  ...createLabOrderMutation(),
  onSuccess: () => { invalidate(); setShowCreate(false); setForm({ labName: '', description: '', expectedDeliveryDate: '' }); },
});
```
