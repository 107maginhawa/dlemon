    ```typescript
    import type { DatabaseInstance } from '@/core/database';
    import { eq } from 'drizzle-orm';
    import { imagingFindings } from './imaging_finding.schema';
    import type { NewImagingFinding, ImagingFinding } from './imaging_finding.schema';

    export class ImagingFindingRepository {
      constructor(private db: DatabaseInstance) {}

      async create(data: NewImagingFinding): Promise<ImagingFinding> {
        const [row] = await this.db.insert(imagingFindings).values(data).returning();
        return row!;
      }

      async findById(id: string): Promise<ImagingFinding | null> {
        const [row] = await this.db.select().from(imagingFindings).where(eq(imagingFindings.id, id)).limit(1);
        return row ?? null;
      }

      async listByImage(imageId: string): Promise<ImagingFinding[]> {
        return this.db.select().from(imagingFindings).where(eq(imagingFindings.imageId, imageId));
      }

      async update(id: string, data: Partial<NewImagingFinding>): Promise<ImagingFinding | null> {
        const [row] = await this.db.update(imagingFindings).set({ ...data, updatedAt: new Date() }).where(eq(imagingFindings.id, id)).returning();
        return row ?? null;
      }

      async delete(id: string): Promise<void> {
        await this.db.delete(imagingFindings).where(eq(imagingFindings.id, id));
      }
    }
    ```
