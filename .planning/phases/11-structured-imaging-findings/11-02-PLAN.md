```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { apiClient } from '@/lib/api-client'   // or equivalent SDK client

export function useMeasurements(imageId: string) {
  const queryClient = useQueryClient()
  const query = useQuery({
    queryKey: ['measurements', imageId],
    queryFn: () => apiClient.get(`/dental/imaging/images/${imageId}/measurements`).then(r => r.json()),
    enabled: !!imageId,
  })
  const create = useMutation({ ... })
  const remove = useMutation({ ... })
  return { measurements: query.data?.items ?? [], isLoading: query.isLoading, create, remove }
}
```
