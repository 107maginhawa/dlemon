    ```typescript
    }
    ```

    **Component state:**
    ```typescript
    const [checks, setChecks] = useState<CheckResult[]>([]);
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState('');
    ```

    **Promise.all on open** (useEffect when `open` changes to true):
    ```typescript
    useEffect(() => {
      if (!open) return;
      setLoading(true);
      setChecks([]);
      setError('');
      Promise.all([
        // 1. Consent: listConsentForms({ path: { visitId } })
        //    pass = result has at least 1 item with signed status
        // 2. Treatments: listDentalTreatments({ path: { visitId } })
        //    pass = no items with status 'diagnosed' or 'planned'
        //    message = `${count} treatment(s) still diagnosed or planned`
        // 3. SOAP notes: getVisitNotes({ path: { visitId } })
        //    pass = at least one of subjective/objective/assessment/plan is non-empty
        //    message = 'SOAP notes not recorded'
        // 4. Lab orders: listLabOrders({ path: { visitId } })
        //    pass = no items with status not in ['completed','cancelled']
        //    message = `${count} lab order(s) still open`
      ]).then(results => {
        setChecks(results);
        setLoading(false);
      }).catch(() => {
        setError('Failed to run checks. Please try again.');
        setLoading(false);
      });
    }, [open, visitId]);
    ```

    Use raw SDK fetch functions (not hooks) inside Promise.all since they run imperatively:
    ```typescript
    import { listConsentForms, listDentalTreatments, getVisitNotes, listLabOrders } from '@monobase/sdk-ts/generated';
    ```
    Check that these function names exist in the generated `sdk.gen.ts`. Use whatever the generated file exports. If consent forms are `listDentalConsentForms`, use that name.

    **Complete Visit mutation:**
    ```typescript
    const queryClient = useQueryClient();
    const completeMutation = useMutation({
      ...updateDentalVisitMutation(),
      onSuccess: () => {
        queryClient.invalidateQueries({ queryKey: listDentalVisitsQueryKey({ query: { patientId } }) });
        onCompleted?.();
        onClose();
      },
    ```
